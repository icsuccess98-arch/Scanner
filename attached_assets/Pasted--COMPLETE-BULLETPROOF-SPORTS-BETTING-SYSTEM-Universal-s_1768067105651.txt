class UniversalSpreadHandler:
    """Handles spread extraction and validation universally"""
    
    @staticmethod
    def extract_spread_data(game_data: Dict) -> Optional[Dict]:
        """Extract spread data with FULL VALIDATION"""
        away_team = game_data.get('away_team')
        home_team = game_data.get('home_team')
        
        bookmakers = game_data.get('bookmakers', [])
        if not bookmakers:
            return None
        
        bookmaker = bookmakers[0]
        markets = bookmaker.get('markets', [])
        
        spreads_market = next((m for m in markets if m['key'] == 'spreads'), None)
        h2h_market = next((m for m in markets if m['key'] == 'h2h'), None)
        
        if not spreads_market:
            return None
        
        outcomes = spreads_market.get('outcomes', [])
        away_spread_outcome = next((o for o in outcomes if o['name'] == away_team), None)
        home_spread_outcome = next((o for o in outcomes if o['name'] == home_team), None)
        
        if not away_spread_outcome or not home_spread_outcome:
            logger.error(f"Missing spread data for {away_team} @ {home_team}")
            return None
        
        away_spread_raw = float(away_spread_outcome.get('point', 0))
        home_spread_raw = float(home_spread_outcome.get('point', 0))
        away_odds = away_spread_outcome.get('price', -110)
        home_odds = home_spread_outcome.get('price', -110)
        
        # Get moneylines
        away_ml = None
        home_ml = None
        if h2h_market:
            h2h_outcomes = h2h_market.get('outcomes', [])
            away_ml_outcome = next((o for o in h2h_outcomes if o['name'] == away_team), None)
            home_ml_outcome = next((o for o in h2h_outcomes if o['name'] == home_team), None)
            if away_ml_outcome:
                away_ml = away_ml_outcome.get('price')
            if home_ml_outcome:
                home_ml = home_ml_outcome.get('price')
        
        # VALIDATE: Spreads are mirror images
        if abs(away_spread_raw + home_spread_raw) > 0.1:
            logger.error(f"SPREAD MISMATCH: {away_team} @ {home_team}")
            return None
        
        # Determine favorite using moneyline
        if away_ml and home_ml:
            if away_ml < home_ml:
                favorite_team = away_team
                underdog_team = home_team
                favorite_location = 'away'
            else:
                favorite_team = home_team
                underdog_team = away_team
                favorite_location = 'home'
        else:
            if away_spread_raw < 0:
                favorite_team = away_team
                underdog_team = home_team
                favorite_location = 'away'
            else:
                favorite_team = home_team
                underdog_team = away_team
                favorite_location = 'home'
        
        spread_data = {
            'away_team': away_team,
            'home_team': home_team,
            'away_spread': away_spread_raw,
            'home_spread': home_spread_raw,
            'away_odds': away_odds,
            'home_odds': home_odds,
            'away_moneyline': away_ml,
            'home_moneyline': home_ml,
            'favorite': favorite_team,
            'underdog': underdog_team,
            'favorite_location': favorite_location,
            'favorite_spread': away_spread_raw if favorite_location == 'away' else home_spread_raw,
            'underdog_spread': home_spread_raw if favorite_location == 'away' else away_spread_raw,
            'spread_magnitude': abs(away_spread_raw),
        }
        
        # VALIDATE: Cross-check with moneyline
        if away_ml and home_ml:
            ml_says_away_fav = away_ml < home_ml
            spread_says_away_fav = away_spread_raw < 0
            
            if ml_says_away_fav != spread_says_away_fav:
                logger.error(f"🚨 SPREAD/MONEYLINE MISMATCH: {away_team} @ {home_team}")
                return None
        
        return spread_data

class TeamNameMatcher:
    """Fuzzy team name matching"""
    
    SUFFIXES = [
        'Eagles', 'Cardinals', 'Wildcats', 'Bulldogs', 'Tigers',
        'Spartans', 'Trojans', 'Bears', 'Panthers', 'Huskies',
        'Gamecocks', 'Volunteers', 'Crimson Tide', 'Blue Devils',
        'Tar Heels', 'Jayhawks', 'Hoosiers', 'Longhorns',
        'State', 'University', 'College', 'Tech'
    ]
    
    EXPANSIONS = {
        'St.': 'Saint', 'St ': 'Saint ',
        'N.C.': 'North Carolina', 'S.C.': 'South Carolina',
        'LA': 'Louisiana', 'USC': 'Southern California',
        'LSU': 'Louisiana State', 'UCLA': 'UC Los Angeles',
    }
    
    @classmethod
    def normalize(cls, team_name: str) -> str:
        """Normalize team name"""
        normalized = team_name
        for abbrev, expansion in cls.EXPANSIONS.items():
            normalized = normalized.replace(abbrev, expansion)
        for suffix in cls.SUFFIXES:
            if normalized.endswith(f' {suffix}'):
                normalized = normalized[:-len(suffix)-1]
        return normalized.strip().lower()
    
    @classmethod
    def match(cls, name1: str, name2: str, threshold: float = 0.85) -> bool:
        """Fuzzy match two team names"""
        norm1 = cls.normalize(name1)
        norm2 = cls.normalize(name2)
        
        if norm1 == norm2:
            return True
        
        ratio = SequenceMatcher(None, norm1, norm2).ratio()
        return ratio >= threshold

class VigAndEVCalculator:
    """Handles all vig removal and EV calculations"""
    
    @staticmethod
    def american_to_probability(odds: float) -> float:
        """Convert American odds to implied probability"""
        if odds < 0:
            return abs(odds) / (abs(odds) + 100)
        else:
            return 100 / (odds + 100)
    
    @staticmethod
    def remove_vig(odds_a: float, odds_b: float) -> Dict:
        """
        Remove vig from two-way market
        
        Returns true probabilities (normalized to 100%)
        """
        prob_a_with_vig = VigAndEVCalculator.american_to_probability(odds_a)
        prob_b_with_vig = VigAndEVCalculator.american_to_probability(odds_b)
        
        total = prob_a_with_vig + prob_b_with_vig
        
        true_prob_a = prob_a_with_vig / total
        true_prob_b = prob_b_with_vig / total
        
        vig_pct = (total - 1.0) * 100
        
        return {
            'true_prob_a': true_prob_a,
            'true_prob_b': true_prob_b,
            'vig_percentage': vig_pct,
            'prob_a_with_vig': prob_a_with_vig,
            'prob_b_with_vig': prob_b_with_vig,
        }
    
    @staticmethod
    def calculate_ev(
        your_probability: float,
        odds: float,
        use_vig_removed: bool = True
    ) -> Dict:
        """
        Calculate Expected Value
        
        Args:
            your_probability: Your model's win probability (0-1)
            odds: American odds
            use_vig_removed: If True, compare against vig-removed market probability
        
        Returns:
            Dict with EV, edge, and recommendation
        """
        # Convert odds to payout multiplier
        if odds < 0:
            payout_multiplier = 100 / abs(odds)
        else:
            payout_multiplier = odds / 100
        
        # Calculate EV
        # EV = (win_prob * payout) - (lose_prob * stake)
        ev = (your_probability * payout_multiplier) - ((1 - your_probability) * 1)
        ev_pct = ev * 100
        
        # Calculate market probability
        market_prob_with_vig = VigAndEVCalculator.american_to_probability(odds)
        
        # Remove vig for true edge calculation
        if use_vig_removed:
            # Assume standard -110/-110 market for vig removal
            market_true_prob = market_prob_with_vig / 1.0476  # Standard 4.76% vig
        else:
            market_true_prob = market_prob_with_vig
        
        # Calculate edge
        edge = your_probability - market_true_prob
        edge_pct = edge * 100
        
        # Determine quality
        if ev_pct >= 5.0:
            quality = 'EXCELLENT'
        elif ev_pct >= 3.0:
            quality = 'GOOD'
        elif ev_pct >= 1.0:
            quality = 'FAIR'
        elif ev_pct >= 0:
            quality = 'MARGINAL'
        else:
            quality = 'NEGATIVE'
        
        return {
            'expected_value': ev,
            'ev_percentage': ev_pct,
            'edge': edge,
            'edge_percentage': edge_pct,
            'your_probability': your_probability,
            'market_true_prob': market_true_prob,
            'market_prob_with_vig': market_prob_with_vig,
            'quality': quality,
            'should_bet': ev > 0
        }

class InjuryImpactCalculator:
    """Calculates injury impact on spreads"""
    
    # Position importance multipliers
    POSITION_WEIGHTS = {
        'PG': 1.2,  # Point guards have high impact
        'SG': 1.0,
        'SF': 1.0,
        'PF': 1.0,
        'C': 1.15,  # Centers important in college basketball
    }
    
    @staticmethod
    def calculate_player_impact(player: Dict) -> float:
        """
        Calculate a single player's impact on the spread
        
        Args:
            player: Dict with keys: ppg, mpg, position, usage_rate, status
        
        Returns:
            Impact in points (positive = helps team, negative = hurts team)
        """
        if player.get('status') != 'OUT':
            return 0.0  # Only count players who are OUT
        
        ppg = player.get('ppg', 0)
        mpg = player.get('mpg', 0)
        position = player.get('position', 'SF')
        usage_rate = player.get('usage_rate', 20)
        
        # Base impact from scoring
        if ppg >= 20:
            base_impact = 4.0  # Star player
        elif ppg >= 15:
            base_impact = 2.5  # Key starter
        elif ppg >= 10:
            base_impact = 1.5  # Role player
        elif ppg >= 5:
            base_impact = 0.5  # Bench player
        else:
            base_impact = 0.2  # Deep bench
        
        # Adjust for minutes played
        minutes_factor = min(mpg / 30, 1.2)  # Cap at 1.2x for 36+ min players
        base_impact *= minutes_factor
        
        # Adjust for position
        position_weight = InjuryImpactCalculator.POSITION_WEIGHTS.get(position, 1.0)
        base_impact *= position_weight
        
        # Adjust for usage rate (high usage players more important)
        if usage_rate >= 30:
            base_impact *= 1.3
        elif usage_rate >= 25:
            base_impact *= 1.15
        
        return base_impact
    
    @staticmethod
    def calculate_team_injury_impact(injuries: List[Dict]) -> Dict:
        """
        Calculate total injury impact for a team
        
        Args:
            injuries: List of player injury dicts
        
        Returns:
            Dict with total impact and breakdown
        """
        if not injuries:
            return {
                'total_impact': 0.0,
                'num_injuries': 0,
                'severity': 'NONE',
                'players_out': []
            }
        
        total_impact = 0.0
        players_out = []
        
        for injury in injuries:
            impact = InjuryImpactCalculator.calculate_player_impact(injury)
            total_impact += impact
            
            if impact > 0:
                players_out.append({
                    'name': injury.get('name', 'Unknown'),
                    'ppg': injury.get('ppg', 0),
                    'impact': round(impact, 1)
                })
        
        # Determine severity
        if total_impact >= 6.0:
            severity = 'CRITICAL'  # Multiple stars or 1 superstar
        elif total_impact >= 4.0:
            severity = 'MAJOR'  # Star player
        elif total_impact >= 2.0:
            severity = 'MODERATE'  # Key starter
        elif total_impact >= 0.5:
            severity = 'MINOR'  # Role player
        else:
            severity = 'MINIMAL'  # Deep bench
        
        return {
            'total_impact': round(total_impact, 2),
            'num_injuries': len([p for p in injuries if p.get('status') == 'OUT']),
            'severity': severity,
            'players_out': sorted(players_out, key=lambda x: x['impact'], reverse=True)
        }
    
    @staticmethod
    def adjust_spread_for_injuries(
        original_spread: float,
        favorite_team: str,
        away_team: str,
        away_injuries: List[Dict],
        home_injuries: List[Dict]
    ) -> Dict:
        """
        Adjust spread based on injuries
        
        Returns adjusted spread and explanation
        """
        away_impact = InjuryImpactCalculator.calculate_team_injury_impact(away_injuries)
        home_impact = InjuryImpactCalculator.calculate_team_injury_impact(home_injuries)
        
        # Net impact (positive = helps favorite, negative = hurts favorite)
        if favorite_team == away_team:
            # Away team is favorite, their injuries hurt the spread
            net_impact = -(away_impact['total_impact']) + home_impact['total_impact']
        else:
            # Home team is favorite, their injuries hurt the spread
            net_impact = away_impact['total_impact'] - home_impact['total_impact']
        
        adjusted_spread = original_spread + net_impact
        
        return {
            'original_spread': original_spread,
            'adjusted_spread': round(adjusted_spread, 1),
            'injury_adjustment': round(net_impact, 1),
            'away_injury_impact': away_impact,
            'home_injury_impact': home_impact,
            'significant_adjustment': abs(net_impact) >= 2.0
        }

class ModelValidator:
    """Validates that all models are calculated correctly"""
    
    @staticmethod
    def validate_spread_model(game: Dict, model_result: Dict) -> Dict:
        """
        Validate spread model calculation
        
        Checks:
        1. Edge calculation is correct
        2. Historical margin meets thresholds
        3. Form validation is correct
        4. EV is calculated properly
        """
        validation = {
            'model_name': 'SPREAD_MODEL',
            'passed': True,
            'errors': [],
            'warnings': []
        }
        
        # Check 1: Edge calculation
        model_spread = model_result.get('predicted_spread')
        market_spread = game['spread']['magnitude']
        expected_edge = abs(model_spread - market_spread)
        actual_edge = model_result.get('edge')
        
        if abs(expected_edge - actual_edge) > 0.1:
            validation['errors'].append(
                f"Edge calculation mismatch: expected {expected_edge:.1f}, got {actual_edge:.1f}"
            )
            validation['passed'] = False
        
        # Check 2: Historical margin threshold (85% for away underdogs)
        if game['spread']['underdog_team'] == game['away_team']:
            margin_pct = model_result.get('historical_margin_pct')
            if margin_pct is not None and margin_pct < 85:
                if model_result.get('spread_history_qualified') == True:
                    validation['errors'].append(
                        f"Away underdog margin {margin_pct}% < 85%, should not qualify"
                    )
                    validation['passed'] = False
        
        # Check 3: EV calculation
        if 'spread_ev' in model_result:
            ev = model_result['spread_ev']
            prob = model_result.get('win_probability', 0.5)
            odds = game['odds']['away'] if game['away_team'] == model_result.get('betting_team') else game['odds']['home']
            
            # Recalculate EV to verify
            calc = VigAndEVCalculator.calculate_ev(prob, odds, use_vig_removed=True)
            if abs(ev - calc['ev_percentage']) > 0.5:
                validation['warnings'].append(
                    f"EV calculation differs: stored {ev:.2f}%, calculated {calc['ev_percentage']:.2f}%"
                )
        
        return validation
    
    @staticmethod
    def validate_total_model(game: Dict, model_result: Dict) -> Dict:
        """Validate totals model calculation"""
        validation = {
            'model_name': 'TOTAL_MODEL',
            'passed': True,
            'errors': [],
            'warnings': []
        }
        
        # Check edge calculation
        predicted_total = model_result.get('predicted_total')
        market_total = model_result.get('market_total')
        expected_edge = abs(predicted_total - market_total)
        actual_edge = model_result.get('edge')
        
        if abs(expected_edge - actual_edge) > 0.1:
            validation['errors'].append(
                f"Edge calculation mismatch: expected {expected_edge:.1f}, got {actual_edge:.1f}"
            )
            validation['passed'] = False
        
        # Check EV if present
        if 'total_ev' in model_result and model_result['total_ev'] is not None:
            ev = model_result['total_ev']
            if ev < -10:  # Sanity check
                validation['warnings'].append(f"Unusually negative EV: {ev:.2f}%")
        
        return validation
    
    @staticmethod
    def validate_all_models(game: Dict, model_results: Dict) -> Dict:
        """Validate all model calculations for a game"""
        validations = []
        
        if 'spread' in model_results:
            validations.append(
                ModelValidator.validate_spread_model(game, model_results['spread'])
            )
        
        if 'total' in model_results:
            validations.append(
                ModelValidator.validate_total_model(game, model_results['total'])
            )
        
        all_passed = all(v['passed'] for v in validations)
        
        return {
            'all_passed': all_passed,
            'validations': validations,
            'game': f"{game['away_team']} @ {game['home_team']}"
        }

    """Fetch Pinnacle odds for comparison"""

        'apiKey': api_key,
        'regions': 'us',
        'markets': 'spreads,h2h',

    }
    
    try:

        return response.json()

        logger.error(f"Error fetching Pinnacle: {e}")
        return {}


def find_matching_pinnacle_game(
    bovada_game: Dict,
    pinnacle_data: List[Dict],
    team_matcher: TeamNameMatcher
) -> Optional[Dict]:
    """Find matching Pinnacle game using fuzzy team matching"""
    for pinnacle_game in pinnacle_data:
        pinn_away = pinnacle_game.get('away_team')
        pinn_home = pinnacle_game.get('home_team')
        
        if (team_matcher.match(bovada_game['away_team'], pinn_away) and
            team_matcher.match(bovada_game['home_team'], pinn_home)):
            
            # Extract Pinnacle spread data
            spread_handler = UniversalSpreadHandler()
            pinn_spread = spread_handler.extract_spread_data(pinnacle_game)
            
            if pinn_spread:
                return {
                    'spread': pinn_spread['spread_magnitude'],
                    'away_odds': pinn_spread['away_odds'],
                    'home_odds': pinn_spread['home_odds'],
                    'away_ml': pinn_spread['away_moneyline'],
                    'home_ml': pinn_spread['home_moneyline'],
                }
    
    return None

def filter_qualified_picks(games: List[Dict]) -> List[Dict]:
    """
    Filter games that meet qualification criteria
    
    Uses YOUR existing qualification logic:
    - spread_is_qualified AND spread_history_qualified
    - total_is_qualified AND total_history_qualified
    - EV >= 0 (or NULL for lenient filtering)
    """
    qualified = []
    
    for game in games:
        spread_model = game['models'].get('spread')
        total_model = game['models'].get('total')
        
        # Check spread qualification
        if spread_model:
            spread_qualified = (
                spread_model.get('spread_is_qualified', False) and
                spread_model.get('spread_history_qualified', False) and
                (spread_model.get('spread_ev') is None or spread_model.get('spread_ev', 0) >= 0)
            )
            
            if spread_qualified:
                qualified.append({
                    'type': 'SPREAD',
                    'game': game,
                    'model': spread_model,
                    'confidence': determine_confidence(game, spread_model)
                })
        
        # Check total qualification
        if total_model:
            total_qualified = (
                total_model.get('total_is_qualified', False) and
                total_model.get('total_history_qualified', False) and
                (total_model.get('total_ev') is None or total_model.get('total_ev', 0) >= 0)
            )
            
            if total_qualified:
                qualified.append({
                    'type': 'TOTAL',
                    'game': game,
                    'model': total_model,
                    'confidence': determine_confidence(game, total_model)
                })
    
    return qualified


def determine_confidence(game: Dict, model: Dict) -> str:
    """
    Determine confidence level based on:
    - Edge magnitude
    - EV quality
    - Injury situation
    - Pinnacle confirmation
    """
    edge = model.get('edge', 0)
    ev = model.get('spread_ev') or model.get('total_ev', 0)
    injury_impact = game['injuries']['adjustment']['injury_adjustment']
    has_pinnacle = game.get('has_pinnacle', False)
    
    # SUPERMAX: High edge + High EV + No major injuries against you
    if edge >= 10 and ev >= 5 and abs(injury_impact) < 3:
        return 'SUPERMAX'
    
    # HIGH: Good edge + Good EV
    elif edge >= 7 and ev >= 3:
        return 'HIGH'
    
    # MEDIUM: Decent edge + Positive EV
    elif edge >= 5 and ev >= 1:
        return 'MEDIUM'
    
    # LOW: Meets minimums
    else:
        return 'LOW'


def rank_picks(qualified_picks: List[Dict]) -> Dict:
    """
    Rank picks by confidence level
    
    Returns picks organized by confidence tier
    """
    ranked = {
        'SUPERMAX': [],
        'HIGH': [],
        'MEDIUM': [],
        'LOW': []
    }
    
    for pick in qualified_picks:
        confidence = pick['confidence']
        ranked[confidence].append(pick)
    
    # Sort each tier by EV
    for tier in ranked:
        ranked[tier].sort(
            key=lambda x: x['model'].get('spread_ev') or x['model'].get('total_ev', 0),
            reverse=True
        )
    
    return ranked
