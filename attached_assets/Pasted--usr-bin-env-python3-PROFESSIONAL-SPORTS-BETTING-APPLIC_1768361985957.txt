class ProbabilityConverter:
    """Convert between odds formats and probabilities."""
    
    @staticmethod
    def american_to_decimal(american: int) -> float:
        if american > 0:
            return (american / 100) + 1
        return (100 / abs(american)) + 1
    
    @staticmethod
    def american_to_implied(american: int) -> float:
        if american > 0:
            return 100 / (american + 100)
        return abs(american) / (abs(american) + 100)
    
    @staticmethod
    def probability_to_american(prob: float) -> int:
        if prob >= 0.5:
            return int(-100 * prob / (1 - prob))
        return int(100 * (1 - prob) / prob)


class VigRemover:
    """Remove vig to get true probabilities."""
    
    @staticmethod
    def remove_two_way_vig(odds_a: int, odds_b: int) -> Dict[str, float]:
        prob_a_vig = ProbabilityConverter.american_to_implied(odds_a)
        prob_b_vig = ProbabilityConverter.american_to_implied(odds_b)
        
        overround = prob_a_vig + prob_b_vig
        prob_a_true = prob_a_vig / overround
        prob_b_true = prob_b_vig / overround
        vig_pct = (overround - 1.0) * 100
        
        return {
            'prob_a': prob_a_true,
            'prob_b': prob_b_true,
            'vig_pct': vig_pct,
            'overround': overround
        }
    
    @staticmethod
    def calculate_fair_line_totals(line: float, over_odds: int, under_odds: int) -> float:
        vig_data = VigRemover.remove_two_way_vig(over_odds, under_odds)
        prob_over = vig_data['prob_a']
        
        if abs(prob_over - 0.5) < 0.04:
            return line
        
        shade_pct = (prob_over - 0.5) * 100
        adjustment = shade_pct * 0.5
        return round(line - adjustment, 1)


class EVCalculator:
    """Calculate EV using Pinnacle as sharp benchmark."""
    
    @staticmethod
    def calculate_ev_vs_pinnacle(bovada_odds: int, pinnacle_odds: int) -> Optional[Dict]:
        if not bovada_odds or not pinnacle_odds:
            return None
        
        pinn_implied = ProbabilityConverter.american_to_implied(pinnacle_odds)
        true_prob = pinn_implied / 1.025  # Remove 2.5% Pinnacle hold
        
        bovada_decimal = ProbabilityConverter.american_to_decimal(bovada_odds)
        ev = (true_prob * bovada_decimal) - 1
        
        b = bovada_decimal - 1
        kelly = (b * true_prob - (1 - true_prob)) / b
        
        return {
            'ev_pct': round(ev * 100, 2),
            'kelly': round(max(kelly, 0), 4),
            'true_prob': round(true_prob, 4)
        }


@dataclass
class EdgeResult:
    qualified: bool
    direction: Optional[str]
    true_edge: float
    raw_edge: float
    fair_line: float
    posted_line: float
    vig_pct: float
    market_balance: str
    confidence: str


class SharpEdgeCalculator:
    """Calculate TRUE edge with vig removal."""
    
    MIN_TRUE_EDGE = {
        'NBA': 3.5,
        'CBB': 4.0,
        'NFL': 2.0,
        'CFB': 2.5,
        'NHL': 0.3
    }
    
    @classmethod
    def calculate_totals_edge(cls, projected_total: float, line: float,
                             over_odds: int, under_odds: int, 
                             league: str) -> EdgeResult:
        
        fair_line = VigRemover.calculate_fair_line_totals(line, over_odds, under_odds)
        vig_data = VigRemover.remove_two_way_vig(over_odds, under_odds)
        
        raw_edge = abs(projected_total - line)
        true_edge = abs(projected_total - fair_line)
        
        direction = 'OVER' if projected_total > fair_line else 'UNDER' if projected_total < fair_line else None
        
        threshold = cls.MIN_TRUE_EDGE.get(league, 3.5)
        qualified = true_edge >= threshold
        
        over_prob = vig_data['prob_a']
        if over_prob > 0.54:
            balance = 'OVER_SHADED'
        elif over_prob < 0.46:
            balance = 'UNDER_SHADED'
        else:
            balance = 'BALANCED'
        
        if true_edge >= threshold * 2.0:
            confidence = 'ELITE'
        elif true_edge >= threshold * 1.5:
            confidence = 'HIGH'
        elif true_edge >= threshold * 1.2:
            confidence = 'STANDARD'
        else:
            confidence = 'LOW'
        
        return EdgeResult(
            qualified=qualified,
            direction=direction,
            true_edge=round(true_edge, 2),
            raw_edge=round(raw_edge, 2),
            fair_line=round(fair_line, 1),
            posted_line=line,
            vig_pct=round(vig_data['vig_pct'], 2),
            market_balance=balance,
            confidence=confidence if qualified else 'NONE'
        )


@dataclass
class QualificationResult:
    qualified: bool
    confidence: str
    bet_size_pct: float
    true_edge: float
    ev_pct: Optional[float]
    reasons_pass: List[str]
    reasons_fail: List[str]

class ProfessionalQualifier:
    """Multi-filter qualification - ALL must pass."""
    
    MIN_EV = 1.0
    MIN_KELLY = 0.02
    MIN_WIN_RATE = 0.58
    MIN_SAMPLE_SIZE = 10  # CRITICAL: Minimum games required
    MAX_VIG = 8.0
    FRACTIONAL_KELLY = 0.25
    MAX_BET_SIZE = 0.10
    
    @classmethod
    def qualify_pick(cls, edge_result: EdgeResult, ev_data: Optional[Dict],
                    historical_win_rate: float, sample_size: int,
                    vig_pct: float, league: str) -> QualificationResult:
        
        reasons_pass = []
        reasons_fail = []
        
        # FILTER 1: True Edge
        min_edge = SharpEdgeCalculator.MIN_TRUE_EDGE[league]
        if edge_result.qualified:
            reasons_pass.append(f"✓ Edge {edge_result.true_edge:.1f} ≥ {min_edge:.1f}")
        else:
            reasons_fail.append(f"✗ Edge {edge_result.true_edge:.1f} < {min_edge:.1f}")
        
        # FILTER 2: EV
        if ev_data is None:
            reasons_pass.append("✓ EV: No Pinnacle (allowed)")
            ev_pct = None
        elif ev_data['ev_pct'] >= cls.MIN_EV:
            reasons_pass.append(f"✓ EV +{ev_data['ev_pct']:.2f}%")
            ev_pct = ev_data['ev_pct']
        else:
            reasons_fail.append(f"✗ EV {ev_data['ev_pct']:+.2f}%")
            ev_pct = ev_data['ev_pct']
        
        # FILTER 3: Kelly
        kelly = ev_data['kelly'] if ev_data else 0
        if kelly >= cls.MIN_KELLY:
            reasons_pass.append(f"✓ Kelly {kelly*100:.2f}%")
        else:
            reasons_fail.append(f"✗ Kelly {kelly*100:.2f}%")
        
        # FILTER 4: History (FIXED: Now considers sample size)
        if historical_win_rate >= cls.MIN_WIN_RATE:
            reasons_pass.append(f"✓ History {historical_win_rate:.1%}")
        else:
            reasons_fail.append(f"✗ History {historical_win_rate:.1%}")
        
        # FILTER 5: Sample Size (CRITICAL FIX)
        if sample_size >= cls.MIN_SAMPLE_SIZE:
            reasons_pass.append(f"✓ Sample {sample_size} games")
        else:
            reasons_fail.append(f"✗ Sample {sample_size} < {cls.MIN_SAMPLE_SIZE}")
        
        # FILTER 6: Vig
        if vig_pct <= cls.MAX_VIG:
            reasons_pass.append(f"✓ Vig {vig_pct:.1f}%")
        else:
            reasons_fail.append(f"✗ Vig {vig_pct:.1f}%")
        
        qualified = len(reasons_fail) == 0
        
        bet_size = 0.0
        if qualified and kelly > 0:
            bet_size = min(kelly * cls.FRACTIONAL_KELLY, cls.MAX_BET_SIZE)
        
        if not qualified:
            confidence = 'NONE'
            recommendation = 'PASS'
        elif edge_result.confidence == 'ELITE' and ev_pct and ev_pct >= 5.0:
            confidence = 'ELITE'
            recommendation = 'MAX_BET'
        elif edge_result.confidence in ['ELITE', 'HIGH'] and ev_pct and ev_pct >= 3.0:
            confidence = 'HIGH'
            recommendation = 'STANDARD_BET'
        else:
            confidence = 'STANDARD'
            recommendation = 'STANDARD_BET'
        
        return QualificationResult(
            qualified=qualified,
            confidence=confidence,
            bet_size_pct=round(bet_size * 100, 2),
            true_edge=edge_result.true_edge,
            ev_pct=ev_pct,
            reasons_pass=reasons_pass,
            reasons_fail=reasons_fail,
            recommendation=recommendation,
            sample_size=sample_size
        )

class TTLCache:
    def __init__(self, maxsize=1000, ttl=3600):
        from collections import OrderedDict
        self.cache = OrderedDict()
        self.maxsize = maxsize
        self.ttl = ttl
    
    def get(self, key):
        if key in self.cache:
            value, timestamp = self.cache[key]
            if time.time() - timestamp < self.ttl:
                return value
            del self.cache[key]
        return None
    
    def set(self, key, value):
        if len(self.cache) >= self.maxsize:
            self.cache.popitem(last=False)
        self.cache[key] = (value, time.time())


espn_schedule_cache = TTLCache(maxsize=500, ttl=43200)
opening_lines_store = TTLCache(maxsize=500, ttl=86400)


class SmartRateLimiter:
    """Rate limiter with exponential backoff."""
    
    def __init__(self, requests_per_second=5):
        self.rate = requests_per_second
        self.tokens = float(requests_per_second)
        self.last_update = time.time()
        self.lock = threading.Lock()
        self.failure_count = 0
    
    def acquire(self):
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            
            if self.failure_count > 0:
                backoff = min(2 ** self.failure_count, 60)
                time.sleep(backoff)
                self.failure_count = max(0, self.failure_count - 1)
            
            self.tokens = min(self.rate, self.tokens + elapsed * self.rate)
            self.last_update = now
            
            if self.tokens >= 1:
                self.tokens -= 1
                return True
            
            sleep_time = (1 - self.tokens) / self.rate
            time.sleep(sleep_time)
            self.tokens = 0
            return True
    
    def record_failure(self):
        with self.lock:
            self.failure_count += 1
    
    def record_success(self):
        with self.lock:
            self.failure_count = max(0, self.failure_count - 1)


espn_limiter = SmartRateLimiter(requests_per_second=5)
odds_limiter = SmartRateLimiter(requests_per_second=2)

def normalize_team_name(name: str) -> str:
    if not name:
        return ""
    n = name.lower().replace("'", "").replace("-", " ").replace(".", "").strip()
    return " ".join(n.split())


def teams_match(name1: str, name2: str) -> bool:
    if not name1 or not name2:
        return False
    
    n1 = normalize_team_name(name1)
    n2 = normalize_team_name(name2)
    
    if n1 == n2:
        return True
    if n1 in n2 or n2 in n1:
        return True
    
    words1 = set(n1.split())
    words2 = set(n2.split())
    
    if not words1 or not words2:
        return False
    
    overlap = words1 & words2
    return len(overlap) >= min(len(words1), len(words2))

def calculate_expected_scores(away_ppg: float, away_opp: float, 
                              home_ppg: float, home_opp: float) -> Tuple[float, float, float]:
    if any(v is None or v < 0 for v in [away_ppg, away_opp, home_ppg, home_opp]):
        raise ValueError("Insufficient data")
    
    exp_away = (away_ppg + home_opp) / 2
    exp_home = (home_ppg + away_opp) / 2
    return exp_away, exp_home, exp_away + exp_home

    }

def calculate_ou_hit_rate_FIXED(games: list, direction: str, 
                               current_line: float = None) -> dict:
    """Calculate O/U hit rate with PROPER push handling."""
    if len(games) < 5:
        return {
            'hit_rate': 0.0,
            'sample_size': 0,
            'valid': False
        }
    
    compare_line = current_line if current_line else sum(g["total"] for g in games) / len(games)
    
    hits = 0
    pushes = 0
    
    for g in games:

        diff = total - compare_line
        
        if abs(diff) < 0.5:  # Push
            pushes += 1
            continue
        
        if direction == "O" and diff > 0:
            hits += 1
        elif direction == "U" and diff < 0:

    if non_push_games == 0:
        return {
            'hit_rate': 0.0,
            'sample_size': 0,
            'valid': False
        }
    
    hit_rate = (hits / non_push_games) * 100
    
    return {
        'hit_rate': round(hit_rate, 1),
        'sample_size': non_push_games,  # CRITICAL: Actual sample size
        'hits': hits,
        'pushes': pushes,
        'valid': True
    }
