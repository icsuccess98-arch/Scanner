pythondef adjust_for_injuries(game, injury_data):
    """
    Injury impact on spreads (college basketball):
    - Starting PG: -2 to -4 points
    - Star scorer (20+ PPG): -3 to -6 points
    - Key big man: -2 to -3 points

pythonclass BankrollManager:
    """Professional bankroll management"""
    
    def calculate_kelly_bet_size(self, edge_pct, win_prob, odds):
        """
        Kelly Criterion: f* = (bp - q) / b
        where:
        - b = odds (decimal - 1)
        - p = probability of winning
        - q = probability of losing (1 - p)
        """
        b = american_to_decimal(odds) - 1
        p = win_prob / 100
        q = 1 - p
        
        kelly_fraction = (b * p - q) / b
        
        # Use FRACTIONAL KELLY (0.25 - 0.5 of full Kelly)
        # Full Kelly is too aggressive, will blow up bankroll
        conservative_kelly = kelly_fraction * 0.25
        
        return max(0, min(conservative_kelly, 0.05))  # Cap at 5% of bankroll
    
    def recommend_units(self, confidence, edge):
        """
        Unit sizing based on confidence:
        - SUPERMAX (90%+): 3-5 units
        - HIGH (80-89%): 2-3 units  
        - MEDIUM (70-79%): 1-2 units
        - LOW (<70%): 0.5-1 unit
        """
        if confidence >= 90:
            return 4.0
        elif confidence >= 80:
            return 2.5
        elif confidence >= 70:
            return 1.5
        else:
            return 1.0
