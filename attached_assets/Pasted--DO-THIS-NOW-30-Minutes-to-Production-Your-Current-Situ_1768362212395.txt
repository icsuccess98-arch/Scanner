class VigRemover:
    @staticmethod
    def remove_two_way_vig(odds_a: int, odds_b: int) -> dict:
        prob_a = abs(odds_a) / (abs(odds_a) + 100) if odds_a < 0 else 100 / (odds_a + 100)
        prob_b = abs(odds_b) / (abs(odds_b) + 100) if odds_b < 0 else 100 / (odds_b + 100)
        total = prob_a + prob_b
        return {
            'prob_a': prob_a / total,
            'prob_b': prob_b / total,
            'vig_pct': (total - 1.0) * 100
        }
    
    @staticmethod
    def get_fair_line(line: float, over_odds: int, under_odds: int) -> float:
        vig = VigRemover.remove_two_way_vig(over_odds, under_odds)
        prob_over = vig['prob_a']
        if abs(prob_over - 0.5) < 0.04:
            return line
        shade = (prob_over - 0.5) * 100 * 0.5
        return round(line - shade, 1)
```

### Step 3: Fix Push Handling (Around Line 1200)

# FIND your calculate_ou_hit_rate function and REPLACE with:

def calculate_ou_hit_rate(games: list, direction: str, current_line: float = None) -> float:
    """FIXED: Excludes pushes from calculation."""
    
    if len(games) < 5:
        return 0.0
    
    line = current_line if current_line else sum(g["total"] for g in games) / len(games)
    
    hits = 0
    pushes = 0
    
    for g in games:
        diff = g["total"] - line
        
        # CRITICAL: Exclude pushes
        if abs(diff) < 0.5:
            pushes += 1
            continue
        
        if direction == "O" and diff > 0:
            hits += 1
        elif direction == "U" and diff < 0:
            hits += 1
    
    non_push = len(games) - pushes
    
    if non_push == 0:
        return 0.0
    
    return (hits / non_push) * 100  # Percentage of NON-PUSH games
```
