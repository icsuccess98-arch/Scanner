import logging

logger = logging.getLogger(__name__)

class UniversalSpreadHandler:
    """
    Handles spread extraction and validation universally
    Works for ANY team (home or away) without confusion
    """
    
    @staticmethod
    def extract_spread_data(game_data: Dict) -> Dict:
        """
        Extract spread data in a way that CANNOT be wrong
        
        Returns standardized format with BOTH teams' perspectives
        No more guessing which direction is which
        """
        away_team = game_data.get('away_team')
        home_team = game_data.get('home_team')
        
        bookmakers = game_data.get('bookmakers', [])
        if not bookmakers:
            return None
        
        bookmaker = bookmakers[0]
        markets = bookmaker.get('markets', [])
        
        spreads_market = next((m for m in markets if m['key'] == 'spreads'), None)
        h2h_market = next((m for m in markets if m['key'] == 'h2h'), None)
        
        if not spreads_market:
            return None
        
        outcomes = spreads_market.get('outcomes', [])
        
        # Find both teams' spreads
        away_spread_outcome = next((o for o in outcomes if o['name'] == away_team), None)
        home_spread_outcome = next((o for o in outcomes if o['name'] == home_team), None)
        
        if not away_spread_outcome or not home_spread_outcome:
            logger.error(f"Missing spread data for {away_team} @ {home_team}")
            return None
        
        # Extract raw spread values
        away_spread_raw = float(away_spread_outcome.get('point', 0))
        home_spread_raw = float(home_spread_outcome.get('point', 0))
        away_odds = away_spread_outcome.get('price', -110)
        home_odds = home_spread_outcome.get('price', -110)
        
        # Get moneylines for validation
        away_ml = None
        home_ml = None
        if h2h_market:
            h2h_outcomes = h2h_market.get('outcomes', [])
            away_ml_outcome = next((o for o in h2h_outcomes if o['name'] == away_team), None)
            home_ml_outcome = next((o for o in h2h_outcomes if o['name'] == home_team), None)
            if away_ml_outcome:
                away_ml = away_ml_outcome.get('price')
            if home_ml_outcome:
                home_ml = home_ml_outcome.get('price')
        
        # ⚠️ CRITICAL VALIDATION: Spreads should be mirror images
        # If away is +7.5, home should be -7.5 (and vice versa)
        if abs(away_spread_raw + home_spread_raw) > 0.1:  # Allow tiny floating point error
            logger.error(
                f"SPREAD MISMATCH: {away_team} @ {home_team}\n"
                f"  Away spread: {away_spread_raw}\n"
                f"  Home spread: {home_spread_raw}\n"
                f"  These should be mirror images!"
            )
            return None
        
        # Determine favorite and underdog using MONEYLINE (most reliable)
        if away_ml and home_ml:
            if away_ml < home_ml:
                favorite_team = away_team
                underdog_team = home_team
                favorite_location = 'away'
            else:
                favorite_team = home_team
                underdog_team = away_team
                favorite_location = 'home'
        else:
            # Fallback to spread if no moneyline (negative spread = favorite)
            if away_spread_raw < 0:
                favorite_team = away_team
                underdog_team = home_team
                favorite_location = 'away'
            else:
                favorite_team = home_team
                underdog_team = away_team
                favorite_location = 'home'
        
        # Build comprehensive spread data - NO AMBIGUITY
        spread_data = {
            # Raw data
            'away_team': away_team,
            'home_team': home_team,
            'away_spread': away_spread_raw,
            'home_spread': home_spread_raw,
            'away_odds': away_odds,
            'home_odds': home_odds,
            'away_moneyline': away_ml,
            'home_moneyline': home_ml,
            
            # Favorite/Underdog (clear labels)
            'favorite': favorite_team,
            'underdog': underdog_team,
            'favorite_location': favorite_location,  # 'away' or 'home'
            'favorite_spread': away_spread_raw if favorite_location == 'away' else home_spread_raw,
            'underdog_spread': home_spread_raw if favorite_location == 'away' else away_spread_raw,
            'favorite_odds': away_odds if favorite_location == 'away' else home_odds,
            'underdog_odds': home_odds if favorite_location == 'away' else away_odds,
            
            # Standard format (spread magnitude, always positive)
            'spread_magnitude': abs(away_spread_raw),
            
            # For database storage (use whatever convention you prefer)
            'spread_away_perspective': away_spread_raw,  # Positive = away dog
            'spread_home_perspective': home_spread_raw,  # Positive = home dog
        }
        
        # ⚠️ VALIDATE: Cross-check spread direction with moneyline
        if away_ml and home_ml:
            ml_says_away_fav = away_ml < home_ml
            spread_says_away_fav = away_spread_raw < 0
            
            if ml_says_away_fav != spread_says_away_fav:
                logger.error(
                    f"🚨 SPREAD/MONEYLINE MISMATCH: {away_team} @ {home_team}\n"
                    f"  Moneyline says: {'Away favorite' if ml_says_away_fav else 'Home favorite'}\n"
                    f"  Spread says: {'Away favorite' if spread_says_away_fav else 'Home favorite'}\n"
                    f"  Away ML: {away_ml}, Home ML: {home_ml}\n"
                    f"  Away Spread: {away_spread_raw}, Home Spread: {home_spread_raw}\n"
                    f"  ❌ DATA IS CORRUPTED - SKIPPING THIS GAME"
                )
                return None
        
        return spread_data
    
    @staticmethod
    def format_spread_for_display(spread_data: Dict, team: str) -> str:
        """
        Format spread for display - NO CONFUSION
        
        Args:
            spread_data: Output from extract_spread_data()
            team: Team name to show spread for
        
        Returns:
            Human-readable string like "Boston College +21.5"
        """
        if team == spread_data['favorite']:
            spread_value = spread_data['favorite_spread']
            odds = spread_data['favorite_odds']
        else:
            spread_value = spread_data['underdog_spread']
            odds = spread_data['underdog_odds']
        
        # Format with proper sign
        sign = '+' if spread_value > 0 else ''
        return f"{team} {sign}{spread_value} ({odds:+d})"
    
    @staticmethod
    def get_spread_for_bet(spread_data: Dict, betting_on: str) -> Tuple[float, int]:
        """
        Get spread and odds for the team you're betting on
        
        Args:
            spread_data: Output from extract_spread_data()
            betting_on: Team name you want to bet on
        
        Returns:
            (spread, odds) tuple
        """
        if betting_on == spread_data['away_team']:
            return spread_data['away_spread'], spread_data['away_odds']
        elif betting_on == spread_data['home_team']:
            return spread_data['home_spread'], spread_data['home_odds']
        else:
            raise ValueError(f"Team '{betting_on}' not in this game")

class TeamNameMatcher:
    """Fuzzy team name matching that works with any variation"""
    
    # Common team name suffixes to remove
    SUFFIXES = [
        'Eagles', 'Cardinals', 'Wildcats', 'Bulldogs', 'Tigers',
        'Spartans', 'Trojans', 'Bears', 'Panthers', 'Huskies',
        'Gamecocks', 'Volunteers', 'Crimson Tide', 'Blue Devils',
        'Tar Heels', 'Jayhawks', 'Hoosiers', 'Longhorns',
        'State', 'University', 'College', 'Tech'
    ]
    
    # Common abbreviations and their expansions
    EXPANSIONS = {
        'St.': 'Saint',
        'St ': 'Saint ',
        'N.C.': 'North Carolina',
        'S.C.': 'South Carolina',
        'LA': 'Louisiana',
        'USC': 'Southern California',
        'LSU': 'Louisiana State',
        'UCLA': 'UC Los Angeles',
        'UNLV': 'Nevada Las Vegas',
        'UTEP': 'Texas El Paso',
        'VCU': 'Virginia Commonwealth',
        'SMU': 'Southern Methodist',
        'TCU': 'Texas Christian'
    }
    
    @classmethod
    def normalize(cls, team_name: str) -> str:
        """Normalize team name for matching"""
        normalized = team_name
        
        # Expand abbreviations
        for abbrev, expansion in cls.EXPANSIONS.items():
            normalized = normalized.replace(abbrev, expansion)
        
        # Remove suffixes
        for suffix in cls.SUFFIXES:
            # Remove exact suffix match at end
            if normalized.endswith(f' {suffix}'):
                normalized = normalized[:-len(suffix)-1]
        
        return normalized.strip().lower()
    
    @classmethod
    def match(cls, name1: str, name2: str, threshold: float = 0.85) -> bool:
        """
        Fuzzy match two team names
        
        Returns True if names match (with high confidence)
        """
        from difflib import SequenceMatcher
        
        norm1 = cls.normalize(name1)
        norm2 = cls.normalize(name2)
        
        # Exact match after normalization
        if norm1 == norm2:
            return True
        
        # Fuzzy match
        ratio = SequenceMatcher(None, norm1, norm2).ratio()
        if ratio >= threshold:
            return True
        
        # Check if one is substring of other (e.g., "North Carolina" vs "North Carolina State")
        if norm1 in norm2 or norm2 in norm1:
            # Only match if the difference is just a suffix
            longer = norm2 if len(norm2) > len(norm1) else norm1
            shorter = norm1 if len(norm1) < len(norm2) else norm2
            suffix = longer.replace(shorter, '').strip()
            
            # Check if suffix is in our known list
            for known_suffix in cls.SUFFIXES:
                if suffix.lower() == known_suffix.lower():
                    return True
        
        return False
    
    @classmethod
    def find_match(cls, target_team: str, team_list: List[str]) -> Optional[str]:
        """
        Find matching team in a list
        
        Returns the matched team name from the list, or None
        """
        for team in team_list:
            if cls.match(target_team, team):
                return team
        return None

def fetch_games_bulletproof(api_key: str, sport: str = 'basketball_ncaab') -> List[Dict]:
    """
    Fetch games with BULLETPROOF spread handling
    
    NO MORE BUGS:
    - ✅ Works for home or away teams
    - ✅ Validates spread direction against moneyline
    - ✅ Handles team name variations
    - ✅ Clear, unambiguous data structure
    """
    import requests
    
    logger.info("=" * 80)
    logger.info("🔒 BULLETPROOF GAME FETCH - UNIVERSAL SPREAD HANDLING")
    logger.info("=" * 80)
    
    url = f"https://api.the-odds-api.com/v4/sports/{sport}/odds/"
    params = {
        'apiKey': api_key,
        'regions': 'us',
        'markets': 'spreads,h2h',
        'oddsFormat': 'american',
        'bookmakers': 'bovada'
    }
    
    try:
        response = requests.get(url, params=params)
        response.raise_for_status()
        games_data = response.json()
    except Exception as e:
        logger.error(f"API error: {e}")
        return []
    
    handler = UniversalSpreadHandler()
    matcher = TeamNameMatcher()
    
    processed_games = []
    errors = []
    
    for game_data in games_data:
        try:
            # Extract spread data with FULL VALIDATION
            spread_data = handler.extract_spread_data(game_data)
            
            if not spread_data:
                errors.append(f"Could not extract spread for game: {game_data.get('away_team')} @ {game_data.get('home_team')}")
                continue
            
            # Build game object with CRYSTAL CLEAR structure
            game = {
                # Basic info
                'game_id': f"{spread_data['away_team']}@{spread_data['home_team']}",
                'away_team': spread_data['away_team'],
                'home_team': spread_data['home_team'],
                'commence_time': game_data.get('commence_time'),
                
                # Spread data - BOTH PERSPECTIVES (choose what you need)
                'spread': {
                    # By team
                    'away': spread_data['away_spread'],
                    'home': spread_data['home_spread'],
                    
                    # By role
                    'favorite': spread_data['favorite_spread'],
                    'underdog': spread_data['underdog_spread'],
                    
                    # Metadata
                    'magnitude': spread_data['spread_magnitude'],
                    'favorite_team': spread_data['favorite'],
                    'underdog_team': spread_data['underdog'],
                    'favorite_location': spread_data['favorite_location'],
                },
                
                # Odds
                'odds': {
                    'away': spread_data['away_odds'],
                    'home': spread_data['home_odds'],
                    'favorite': spread_data['favorite_odds'],
                    'underdog': spread_data['underdog_odds'],
                },
                
                # Moneylines
                'moneyline': {
                    'away': spread_data['away_moneyline'],
                    'home': spread_data['home_moneyline'],
                },
                
                # Human-readable display strings
                'display': {
                    'away': handler.format_spread_for_display(spread_data, spread_data['away_team']),
                    'home': handler.format_spread_for_display(spread_data, spread_data['home_team']),
                    'favorite': handler.format_spread_for_display(spread_data, spread_data['favorite']),
                    'underdog': handler.format_spread_for_display(spread_data, spread_data['underdog']),
                },
                
                # Validation passed
                'validated': True
            }
            
            processed_games.append(game)
            
            # Log for verification
            logger.info(f"✅ {game['display']['away']} @ {game['display']['home']}")
            logger.info(f"   Favorite: {spread_data['favorite']} ({spread_data['favorite_location']})")
            
        except Exception as e:
            errors.append(f"Error processing game: {e}")
            logger.error(f"Error: {e}")
            continue
    
    # Summary
    logger.info("=" * 80)
    logger.info(f"✅ Successfully processed: {len(processed_games)} games")
    if errors:
        logger.warning(f"⚠️  Errors encountered: {len(errors)}")
        for error in errors:
            logger.warning(f"   {error}")
    logger.info("=" * 80)
    
    return processed_games
