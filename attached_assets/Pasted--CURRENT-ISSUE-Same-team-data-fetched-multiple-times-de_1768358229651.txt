class ESPNDataCache:
    """Intelligent ESPN data caching with prefetching."""
    
    def __init__(self):
        self.schedule_cache = TTLCache(maxsize=500, ttl=43200)  # 12 hours
        self.team_id_cache = {}  # Permanent cache
        self.pending_requests = {}  # Deduplication
        self.request_lock = threading.Lock()
    
    def get_or_fetch_schedule(self, team_name, league, fetch_func):
        """Get schedule with request deduplication."""
        cache_key = f"{league}:{team_name.lower()}"
        
        # Check cache first
        cached = self.schedule_cache.get(cache_key)
        if cached:
            return cached
        
        # Deduplicate concurrent requests for same team
        with self.request_lock:
            if cache_key in self.pending_requests:
                # Another thread is already fetching this
                event = self.pending_requests[cache_key]
            else:
                # We're the first to request this
                event = threading.Event()
                self.pending_requests[cache_key] = event
        
        if event in self.pending_requests.values() and not event.is_set():
            # Wait for other thread to complete
            event.wait(timeout=10)
            cached = self.schedule_cache.get(cache_key)
            if cached:
                return cached
        
        # Fetch data
        try:
            data = fetch_func(team_name, league)
            self.schedule_cache.set(cache_key, data)
            
            # Signal other waiting threads
            with self.request_lock:
                if cache_key in self.pending_requests:
                    self.pending_requests[cache_key].set()
                    del self.pending_requests[cache_key]
            
            return data
        
        except Exception as e:
            # Clean up on error
            with self.request_lock:
                if cache_key in self.pending_requests:
                    self.pending_requests[cache_key].set()
                    del self.pending_requests[cache_key]
            raise
