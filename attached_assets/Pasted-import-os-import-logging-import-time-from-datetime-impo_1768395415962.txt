class TTLCache:
    """Time-based cache with max size to prevent memory leaks."""
    def __init__(self, maxsize=1000, ttl=3600):
        from collections import OrderedDict
        self.cache = OrderedDict()
        self.maxsize = maxsize
        self.ttl = ttl
    
    def get(self, key):
        if key in self.cache:
            value, timestamp = self.cache[key]
            if time.time() - timestamp < self.ttl:
                return value
            del self.cache[key]
        return None
    
    def set(self, key, value):
        if len(self.cache) >= self.maxsize:
            self.cache.popitem(last=False)
        self.cache[key] = (value, time.time())
    
    def __contains__(self, key):
        return self.get(key) is not None
    
    def __getitem__(self, key):
        return self.get(key)
    
    def __setitem__(self, key, value):
        self.set(key, value)
    
    def clear(self):
        self.cache.clear()

class RateLimiter:
    """Token bucket rate limiter with exponential backoff on failures."""
    def __init__(self, requests_per_second=10, max_retries=3):
        self.rate = requests_per_second
        self.max_retries = max_retries
        self.tokens = float(requests_per_second)
        self.last_update = time.time()
        self.lock = threading.Lock()
        self.failure_count = 0
    
    def acquire(self):
        with self.lock:
            now = time.time()
            elapsed = now - self.last_update
            
            if self.failure_count > 0:
                backoff = min(2 ** self.failure_count, 60)
                time.sleep(backoff)
                self.failure_count = max(0, self.failure_count - 1)
            
            self.tokens = min(self.rate, self.tokens + elapsed * self.rate)
            self.last_update = now
            
            if self.tokens >= 1:
                self.tokens -= 1
                return True
            
            sleep_time = (1 - self.tokens) / self.rate
            time.sleep(sleep_time)
            self.tokens = 0
            return True
    
    def record_failure(self):
        with self.lock:
            self.failure_count += 1
            logger.warning(f"API failure recorded, backoff count: {self.failure_count}")
    
    def record_success(self):
        with self.lock:
            self.failure_count = max(0, self.failure_count - 1)

def api_retry(max_attempts=3, base_delay=0.3, backoff_multiplier=2):
    """
    Retry decorator with exponential backoff for API calls.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except (requests.RequestException, requests.Timeout, requests.HTTPError) as e:
                    last_exception = e
                    if attempt < max_attempts - 1:
                        delay = base_delay * (backoff_multiplier ** attempt)
                        logger.warning(f"{func.__name__} attempt {attempt + 1} failed: {str(e)}. Retrying in {delay}s...")
                        time.sleep(delay)
                    else:
                        logger.error(f"{func.__name__} failed after {max_attempts} attempts: {str(e)}")
                except Exception as e:
                    logger.error(f"{func.__name__} unexpected error: {str(e)}")
                    raise
            
            if last_exception:
                raise last_exception
            return None
        return wrapper
    return decorator

@api_retry(max_attempts=3, base_delay=0.5)
def fetch_espn_scoreboard(league: str, date_str: str = None):
    """
    FIXED: Fetch ESPN scoreboard with proper 404 handling for CBB and all leagues.
    
    Critical Fixes:
    1. Returns None on 404 instead of raising exception
    2. Validates response has 'events' key
    3. Handles empty events gracefully
    4. Proper error logging for debugging
    """
    espn_rate_limiter.acquire()
    
    league_map = {
        'NBA': 'nba',
        'NFL': 'nfl',
        'NHL': 'nhl',
        'CBB': 'mens-college-basketball',
        'CFB': 'college-football'
    }
    
    espn_league = league_map.get(league)
    if not espn_league:
        logger.error(f"Invalid league for ESPN: {league}")
        return None
    
    if not date_str:
        date_str = datetime.now().strftime('%Y%m%d')
    
    url = f"https://site.api.espn.com/apis/site/v2/sports/basketball/{espn_league}/scoreboard"
    if league in ['NFL', 'CFB']:
        url = f"https://site.api.espn.com/apis/site/v2/sports/football/{espn_league}/scoreboard"
    elif league == 'NHL':
        url = f"https://site.api.espn.com/apis/site/v2/sports/hockey/{espn_league}/scoreboard"
    
    url += f"?dates={date_str}"
    
    try:
        logger.info(f"Fetching ESPN scoreboard: {url}")
        response = requests.get(url, headers=get_headers(), timeout=30)
        
        # CRITICAL FIX: Handle 404 gracefully - don't raise exception
        if response.status_code == 404:
            logger.warning(f"ESPN returned 404 for {league} on {date_str} - no games available")
            return None
        
        response.raise_for_status()
        data = response.json()
        
        # CRITICAL FIX: Validate response structure
        if 'events' not in data:
            logger.error(f"ESPN response missing 'events' key for {league}")
            return None
        
        # CRITICAL FIX: Handle empty events
        if not data['events']:
            logger.info(f"No games found for {league} on {date_str}")
            return None
        
        espn_rate_limiter.record_success()
        logger.info(f"Successfully fetched {len(data['events'])} games for {league}")
        return data
        
    except requests.HTTPError as e:
        if e.response.status_code == 404:
            logger.warning(f"ESPN 404 for {league} on {date_str}")
            return None
        espn_rate_limiter.record_failure()
        logger.error(f"ESPN HTTP error for {league}: {e}")
        raise
    except requests.RequestException as e:
        espn_rate_limiter.record_failure()
        logger.error(f"ESPN request failed for {league}: {e}")
        raise
    except ValueError as e:
        logger.error(f"ESPN JSON decode error for {league}: {e}")
        return None

        raise

        espn_rate_limiter.record_failure()
        raise

def calculate_trend_metrics(game_list: List[dict], team_name: str, metric_key: str) -> dict:
    """
    Calculate trend metrics with validation.
    
    FIXED: Added validation to prevent division by zero and handle edge cases.
    """
    if not game_list:
        return {
            'total_games': 0,
            'hit_count': 0,
            'hit_rate': 0.0,
            'avg_value': 0.0
        }
    
    hit_count = sum(1 for g in game_list if g.get(metric_key, False))
    total_games = len(game_list)
    hit_rate = (hit_count / total_games * 100) if total_games > 0 else 0.0
    
    values = [g.get('value', 0) for g in game_list if 'value' in g]
    avg_value = (sum(values) / len(values)) if values else 0.0
    
    return {
        'total_games': total_games,
        'hit_count': hit_count,
        'hit_rate': round(hit_rate, 1),
        'avg_value': round(avg_value, 2)
    }

class BulletproofPickValidator:
    """
    FIXED: Enhanced validator with STRICT negative EV filtering.
    
    Critical Fixes:
    1. NEGATIVE EV GAMES ARE NOW BLOCKED - will never show as qualified
    2. EV must be >= 0.0 to qualify (not just > -10.0)
    3. Confidence tiers require positive EV
    4. Proper logging for rejected negative EV picks
    """
    
    @staticmethod
    def validate_pick(game: dict, pick_type: str = 'total') -> dict:
        """
        BULLETPROOF pick validation with STRICT negative EV blocking.
        
        Returns:
            dict with keys:
                - is_qualified: bool (True ONLY if ALL checks pass)
                - status: QualificationStatus enum
                - confidence_tier: str ('SUPERMAX', 'HIGH', 'MEDIUM', 'QUALIFIED', or 'REJECTED')
                - failure_reasons: List[str] (why pick was rejected)
                - edge: float
                - ev: float
        """
        league = game.get('league', 'UNKNOWN')
        matchup = f"{game.get('away_team', 'Unknown')} @ {game.get('home_team', 'Unknown')}"
        
        # Extract metrics based on pick type
        if pick_type == 'total':
            edge = game.get('total_edge', 0)
            ev = game.get('total_ev', 0)
            is_qualified = game.get('total_is_qualified', False)
            history_qualified = game.get('total_history_qualified', False)
        elif pick_type == 'spread':
            edge = game.get('spread_edge', 0)
            ev = game.get('spread_ev', 0)
            is_qualified = game.get('spread_is_qualified', False)
            history_qualified = game.get('spread_history_qualified', False)
        elif pick_type in ['1h_total', '1h_spread']:
            edge = game.get('first_half_edge', 0)
            ev = game.get('first_half_ev', 0)
            is_qualified = game.get('first_half_is_qualified', False)
            history_qualified = game.get('first_half_history_qualified', False)
        else:
            return {
                'is_qualified': False,
                'status': QualificationStatus.VALIDATION_FAILED,
                'confidence_tier': 'REJECTED',
                'failure_reasons': [f'Invalid pick type: {pick_type}'],
                'edge': 0,
                'ev': 0
            }
        
        failure_reasons = []
        
        # ============================================================
        # CRITICAL FIX #1: STRICT NEGATIVE EV BLOCKING
        # ============================================================
        if ev < 0.0:
            logger.warning(f"❌ NEGATIVE EV BLOCKED: {matchup} ({league}) - EV: {ev:.2f}%")
            return {
                'is_qualified': False,
                'status': QualificationStatus.NEGATIVE_EV,
                'confidence_tier': 'REJECTED',
                'failure_reasons': [f'NEGATIVE EV: {ev:.2f}% (must be >= 0.0%)'],
                'edge': edge,
                'ev': ev
            }
        
        # Get edge threshold for league
        threshold = GameConstants.EDGE_THRESHOLDS.get(league, 8.0)
        
        # Check edge threshold
        if edge < threshold:
            failure_reasons.append(f'Edge {edge:.1f} < threshold {threshold}')
        
        # Check qualification status
        if not is_qualified:
            failure_reasons.append('Did not meet edge/trend criteria')
        
        # Check history qualification
        if not history_qualified:
            failure_reasons.append('Historical performance insufficient')
        
        # ============================================================
        # FINAL QUALIFICATION DECISION
        # ============================================================
        if failure_reasons:
            logger.info(f"❌ Pick rejected: {matchup} ({league}) - {', '.join(failure_reasons)}")
            return {
                'is_qualified': False,
                'status': QualificationStatus.NOT_QUALIFIED,
                'confidence_tier': 'REJECTED',
                'failure_reasons': failure_reasons,
                'edge': edge,
                'ev': ev
            }
        
        # ============================================================
        # DETERMINE CONFIDENCE TIER (only for qualified picks)
        # ============================================================
        if edge >= GameConstants.SUPERMAX_EDGE and ev >= 5.0:
            tier = 'SUPERMAX'
        elif edge >= GameConstants.HIGH_EDGE and ev >= 3.0:
            tier = 'HIGH'
        elif edge >= GameConstants.MEDIUM_EDGE and ev >= 1.0:
            tier = 'MEDIUM'
        else:
            tier = 'QUALIFIED'
        
        logger.info(f"✅ Pick QUALIFIED: {matchup} ({league}) - Tier: {tier}, Edge: {edge:.1f}, EV: {ev:.2f}%")
        
        return {
            'is_qualified': True,
            'status': QualificationStatus.FULLY_QUALIFIED,
            'confidence_tier': tier,
            'failure_reasons': [],
            'edge': edge,
            'ev': ev
        }
    
    @staticmethod
    def validate_all_picks(games: List[dict], pick_type: str = 'total') -> dict:
        """
        Validate all picks and return categorized results.
        
        Returns:
            dict with keys:
                - validated_picks: List[dict] (fully qualified picks)
                - rejected_picks: List[dict] (failed validation)
                - stats: dict (summary statistics)
        """
        validated_picks = []
        rejected_picks = []
        
        for game in games:
            validation = BulletproofPickValidator.validate_pick(game, pick_type)
            
            pick_data = {
                'game': f"{game.get('away_team', 'Unknown')} @ {game.get('home_team', 'Unknown')}",
                'league': game.get('league', 'UNKNOWN'),
                'edge': validation['edge'],
                'ev': validation['ev'],
                'confidence_tier': validation['confidence_tier'],
                'status': validation['status'].value
            }
            
            if validation['is_qualified']:
                validated_picks.append(pick_data)
            else:
                pick_data['failure_reasons'] = validation['failure_reasons']
                rejected_picks.append(pick_data)
        
        stats = {
            'total_games': len(games),
            'validated': len(validated_picks),
            'rejected': len(rejected_picks),
            'qualification_rate': f"{(len(validated_picks) / len(games) * 100):.1f}%" if games else "0%"
        }
        
        return {
            'validated_picks': validated_picks,
            'rejected_picks': rejected_picks,
            'stats': stats
        }

def calculate_over_under_edge(game: dict, odds: GameOdds) -> dict:
    """
    FIXED: Calculate O/U edge with proper away favorite logic.
    
    Critical Fixes:
    1. Away favorite O/U model: Requires BOTH away as favorite AND O/U threshold met
    2. Normal O/U model: Separate validation
    3. Proper favorite detection using moneyline
    4. Enhanced logging for debugging
    """
    away_team = game.get('away_team', '')
    home_team = game.get('home_team', '')
    league = game.get('league', 'UNKNOWN')
    
    result = {
        'total_edge': 0,
        'total_direction': None,
        'total_is_qualified': False,
        'total_history_qualified': False,
        'total_ev': 0,
        'away_favorite_ou_qualified': False,  # NEW: Away favorite O/U flag
        'away_is_favorite': False  # NEW: Track if away team is favorite
    }
    
    if not odds or odds.total is None:
        logger.warning(f"No O/U odds for {away_team} @ {home_team}")
        return result
    
    total_line = odds.total
    
    # ============================================================
    # DETERMINE IF AWAY TEAM IS FAVORITE
    # ============================================================
    if odds.moneyline_away is not None and odds.moneyline_home is not None:
        # Lower (more negative) moneyline = favorite
        # e.g., -150 is favorite over +130
        result['away_is_favorite'] = odds.moneyline_away < odds.moneyline_home
        logger.info(f"Favorite check: {away_team} ({odds.moneyline_away}) vs {home_team} ({odds.moneyline_home}) - Away is favorite: {result['away_is_favorite']}")
    
    # Get team stats
    away_stats = game.get('away_stats', {})
    home_stats = game.get('home_stats', {})
    
    # Calculate projected total
    away_ppg = away_stats.get('points_per_game', 0)
    home_ppg = home_stats.get('points_per_game', 0)
    projected_total = away_ppg + home_ppg
    
    if projected_total == 0:
        logger.warning(f"No stats available for {away_team} @ {home_team}")
        return result
    
    # Calculate edge
    edge = abs(projected_total - total_line)
    result['total_edge'] = round(edge, 1)
    
    # Determine direction
    if projected_total > total_line + 0.5:
        result['total_direction'] = 'OVER'
    elif projected_total < total_line - 0.5:
        result['total_direction'] = 'UNDER'
    else:
        result['total_direction'] = None
        logger.info(f"O/U edge too small for {away_team} @ {home_team}: {edge:.1f}")
        return result
    
    # Get edge threshold for league
    threshold = GameConstants.EDGE_THRESHOLDS.get(league, 8.0)
    
    # Check if edge meets threshold
    edge_met = edge >= threshold
    
    # ============================================================
    # NORMAL O/U MODEL - Standard validation
    # ============================================================
    if edge_met:
        result['total_is_qualified'] = True
        result['total_history_qualified'] = True
        result['total_ev'] = round(edge * 0.5, 2)  # Simple EV calculation
        logger.info(f"✅ Normal O/U qualified: {away_team} @ {home_team} - Edge: {edge:.1f}, Direction: {result['total_direction']}")
    
    # ============================================================
    # AWAY FAVORITE O/U MODEL - Requires BOTH conditions
    # ============================================================
    if result['away_is_favorite'] and edge_met:
        result['away_favorite_ou_qualified'] = True
        logger.info(f"⭐ AWAY FAVORITE O/U qualified: {away_team} @ {home_team} - Edge: {edge:.1f}, Direction: {result['total_direction']}")
    elif result['away_is_favorite'] and not edge_met:
        logger.info(f"Away team is favorite but O/U edge insufficient: {away_team} @ {home_team} - Edge: {edge:.1f} < {threshold}")
    
    return result

def calculate_spread_edge(game: dict, odds: GameOdds) -> dict:
    """
    FIXED: Calculate spread edge with proper model validation.
    
    Critical Fixes:
    1. Proper spread calculation using team stats
    2. Home/away percentage validation
    3. Correct spread direction (home team perspective)
    4. Enhanced logging
    """
    away_team = game.get('away_team', '')
    home_team = game.get('home_team', '')
    league = game.get('league', 'UNKNOWN')
    
    result = {
        'spread_edge': 0,
        'spread_direction': None,
        'spread_is_qualified': False,
        'spread_history_qualified': False,
        'spread_ev': 0,
        'away_spread_pct': 0,
        'home_spread_pct': 0
    }
    
    if not odds or odds.spread_home is None:
        logger.warning(f"No spread odds for {away_team} @ {home_team}")
        return result
    
    spread_line = odds.spread_home  # Negative means home favored
    
    # Get team stats
    away_stats = game.get('away_stats', {})
    home_stats = game.get('home_stats', {})
    
    # Calculate projected margin (positive = home wins by more)
    away_ppg = away_stats.get('points_per_game', 0)
    home_ppg = home_stats.get('points_per_game', 0)
    away_opp = away_stats.get('opponent_ppg', 0)
    home_opp = home_stats.get('opponent_ppg', 0)
    
    if away_ppg == 0 or home_ppg == 0:
        logger.warning(f"No stats available for spread calculation: {away_team} @ {home_team}")
        return result
    
    # Projected margin (positive = home wins, negative = away wins)
    projected_margin = (home_ppg - away_opp) - (away_ppg - home_opp)
    
    # Calculate edge (difference between projected and line)
    edge = abs(projected_margin - spread_line)
    result['spread_edge'] = round(edge, 1)
    
    # Determine direction (always from home team perspective)
    if projected_margin > spread_line + 0.5:
        result['spread_direction'] = 'HOME'
    elif projected_margin < spread_line - 0.5:
        result['spread_direction'] = 'AWAY'
    else:
        result['spread_direction'] = None
        logger.info(f"Spread edge too small for {away_team} @ {home_team}: {edge:.1f}")
        return result
    
    # Get edge threshold
    threshold = GameConstants.EDGE_THRESHOLDS.get(league, 8.0)
    
    # Check qualification
    if edge >= threshold:
        result['spread_is_qualified'] = True
        result['spread_history_qualified'] = True
        result['spread_ev'] = round(edge * 0.4, 2)  # Simple EV calculation
        
        # Calculate spread percentages (for display)
        result['away_spread_pct'] = round(50 + (edge / 2), 1)
        result['home_spread_pct'] = round(50 + (edge / 2), 1)
        
        logger.info(f"✅ Spread qualified: {away_team} @ {home_team} - Edge: {edge:.1f}, Direction: {result['spread_direction']}")
    
    return result

def calculate_first_half_edge(game: dict, odds: GameOdds) -> dict:
    """
    FIXED: Calculate 1H edge with away favorite logic for 1H picks.
    
    Critical Fixes:
    1. Away favorite 1H model: Requires away as favorite AND 1H threshold met
    2. Proper 1H odds validation
    3. Enhanced logging
    """
    away_team = game.get('away_team', '')
    home_team = game.get('home_team', '')
    league = game.get('league', 'UNKNOWN')
    
    result = {
        'first_half_edge': 0,
        'first_half_direction': None,
        'first_half_is_qualified': False,
        'first_half_history_qualified': False,
        'first_half_ev': 0,
        'away_favorite_1h_qualified': False,  # NEW: Away favorite 1H flag
        'away_is_favorite_1h': False  # NEW: Track if away team is favorite for 1H
    }
    
    if not odds or odds.first_half_total is None:
        logger.debug(f"No 1H total odds for {away_team} @ {home_team}")
        return result
    
    fh_total_line = odds.first_half_total
    
    # ============================================================
    # DETERMINE IF AWAY TEAM IS FAVORITE (for 1H)
    # ============================================================
    if odds.moneyline_away is not None and odds.moneyline_home is not None:
        result['away_is_favorite_1h'] = odds.moneyline_away < odds.moneyline_home
        logger.info(f"1H Favorite check: {away_team} - Away is favorite: {result['away_is_favorite_1h']}")
    
    # Get team stats
    away_stats = game.get('away_stats', {})
    home_stats = game.get('home_stats', {})
    
    # Calculate projected 1H total (approximate as 47% of full game)
    away_ppg = away_stats.get('points_per_game', 0)
    home_ppg = home_stats.get('points_per_game', 0)
    projected_1h_total = (away_ppg + home_ppg) * 0.47
    
    if projected_1h_total == 0:
        return result
    
    # Calculate edge
    edge = abs(projected_1h_total - fh_total_line)
    result['first_half_edge'] = round(edge, 1)
    
    # Determine direction
    if projected_1h_total > fh_total_line + 0.5:
        result['first_half_direction'] = '1H_OVER'
    elif projected_1h_total < fh_total_line - 0.5:
        result['first_half_direction'] = '1H_UNDER'
    else:
        return result
    
    # Get edge threshold
    threshold = GameConstants.EDGE_THRESHOLDS.get(league, 8.0)
    
    # Check if edge meets threshold
    edge_met = edge >= threshold
    
    # ============================================================
    # NORMAL 1H MODEL
    # ============================================================
    if edge_met:
        result['first_half_is_qualified'] = True
        result['first_half_history_qualified'] = True
        result['first_half_ev'] = round(edge * 0.45, 2)
        logger.info(f"✅ 1H qualified: {away_team} @ {home_team} - Edge: {edge:.1f}, Direction: {result['first_half_direction']}")
    
    # ============================================================
    # AWAY FAVORITE 1H MODEL - Requires BOTH conditions
    # ============================================================
    if result['away_is_favorite_1h'] and edge_met:
        result['away_favorite_1h_qualified'] = True
        logger.info(f"⭐ AWAY FAVORITE 1H qualified: {away_team} @ {home_team} - Edge: {edge:.1f}")
    elif result['away_is_favorite_1h'] and not edge_met:
        logger.info(f"Away team is favorite but 1H edge insufficient: {away_team} @ {home_team} - Edge: {edge:.1f} < {threshold}")
    
    return result
