class ValidationResult:
    """Track validation test results."""
    
    def __init__(self):
        self.passed = 0
        self.failed = 0
        self.warnings = 0
        self.tests = []
    
    def add_test(self, name: str, passed: bool, message: str = "", warning: bool = False):
        """Add a test result."""
        status = "✅ PASS" if passed else "❌ FAIL"
        if warning:
            status = "⚠️  WARN"
            self.warnings += 1
        elif passed:
            self.passed += 1
        else:
            self.failed += 1
        
        self.tests.append({
            'name': name,
            'passed': passed,
            'status': status,
            'message': message
        })
        
        logger.info(f"{status}: {name} - {message}")
    
    def print_summary(self):
        """Print final summary."""
        print("\n" + "=" * 80)
        print("VALIDATION SUMMARY")
        print("=" * 80)
        
        for test in self.tests:
            print(f"{test['status']}: {test['name']}")
            if test['message']:
                print(f"   → {test['message']}")
        
        print("\n" + "-" * 80)
        total = self.passed + self.failed + self.warnings
        print(f"Total Tests: {total}")
        print(f"✅ Passed: {self.passed}")
        print(f"❌ Failed: {self.failed}")
        print(f"⚠️  Warnings: {self.warnings}")
        print("=" * 80)
        
        if self.failed == 0:
            print("\n🎉 ALL CRITICAL TESTS PASSED - System is production-ready!")
        else:
            print(f"\n⚠️  {self.failed} CRITICAL TEST(S) FAILED - Review errors above")
        
        return self.failed == 0

def test_injury_fetching(results: ValidationResult):
    """Test 2: Injury data fetching."""
    print("\n" + "=" * 80)
    print("TEST 2: INJURY DATA FETCHING")
    print("=" * 80)
    
    from rotowire_integration import fetch_team_injuries
    
    test_cases = [
        ("Lakers", "NBA"),
        ("Warriors", "NBA"),
        ("Cowboys", "NFL"),
    ]
    
    for team, league in test_cases:
        try:
            injuries, source = fetch_team_injuries(team, league)
            
            # Validate return types
            if not isinstance(injuries, list):
                results.add_test(
                    f"Fetch injuries: {team} ({league})",
                    False,
                    f"Expected list, got {type(injuries)}"
                )
                continue
            
            if source not in ['rotowire', 'espn', 'none', 'error']:
                results.add_test(
                    f"Fetch injuries: {team} ({league})",
                    False,
                    f"Invalid source: {source}"
                )
                continue
            
            # Success
            results.add_test(
                f"Fetch injuries: {team} ({league})",
                True,
                f"Found {len(injuries)} injuries from {source}"
            )
            
            # Validate injury objects if any exist
            if injuries:
                injury = injuries[0]
                if not hasattr(injury, 'name'):
                    results.add_test(
                        f"Validate injury object: {team}",
                        False,
                        "Injury object missing 'name' attribute"
                    )
                elif not hasattr(injury, 'status'):
                    results.add_test(
                        f"Validate injury object: {team}",
                        False,
                        "Injury object missing 'status' attribute"
                    )
                elif not hasattr(injury, 'impact_score'):
                    results.add_test(
                        f"Validate injury object: {team}",
                        False,
                        "Injury object missing 'impact_score' attribute"
                    )
                else:
                    results.add_test(
                        f"Validate injury object: {team}",
                        True,
                        f"Injury: {injury.name} ({injury.status.label}) - Impact: {injury.impact_score}"
                    )
        
        except Exception as e:
            results.add_test(
                f"Fetch injuries: {team} ({league})",
                False,
                f"Exception: {str(e)}"
            )

def test_cache_functionality(results: ValidationResult):
    """Test 5: Cache functionality."""
    print("\n" + "=" * 80)
    print("TEST 5: CACHE FUNCTIONALITY")
    print("=" * 80)
    
    from rotowire_integration import get_cache_stats, clear_cache, fetch_team_injuries
    
    # Test 1: Get cache stats
    try:
        stats = get_cache_stats()
        
        if not isinstance(stats, dict):
            results.add_test(
                "Cache: Get stats",
                False,
                f"Expected dict, got {type(stats)}"
            )
        elif 'total_entries' not in stats:
            results.add_test(
                "Cache: Get stats",
                False,
                "Stats missing 'total_entries' key"
            )
        elif 'valid_entries' not in stats:
            results.add_test(
                "Cache: Get stats",
                False,
                "Stats missing 'valid_entries' key"
            )
        else:
            results.add_test(
                "Cache: Get stats",
                True,
                f"Total: {stats['total_entries']}, Valid: {stats['valid_entries']}"
            )
    except Exception as e:
        results.add_test(
            "Cache: Get stats",
            False,
            f"Exception: {str(e)}"
        )
    
    # Test 2: Clear cache
    try:
        clear_cache()
        stats_after = get_cache_stats()
        
        if stats_after['total_entries'] == 0:
            results.add_test(
                "Cache: Clear",
                True,
                "Cache cleared successfully"
            )
        else:
            results.add_test(
                "Cache: Clear",
                False,
                f"Cache not empty after clear: {stats_after['total_entries']} entries"
            )
    except Exception as e:
        results.add_test(
            "Cache: Clear",
            False,
            f"Exception: {str(e)}"
        )
    
    # Test 3: Cache hit
    try:
        # First fetch (cache miss)
        injuries1, source1 = fetch_team_injuries("Lakers", "NBA")
        
        # Second fetch (should be cache hit)
        injuries2, source2 = fetch_team_injuries("Lakers", "NBA")
        
        # Both should return same data
        if len(injuries1) == len(injuries2):
            results.add_test(
                "Cache: Hit consistency",
                True,
                f"Cache returned consistent data ({len(injuries1)} injuries)"
            )
        else:
            results.add_test(
                "Cache: Hit consistency",
                False,
                f"Cache data inconsistent: {len(injuries1)} vs {len(injuries2)}"
            )
    except Exception as e:
        results.add_test(
            "Cache: Hit consistency",
            False,
            f"Exception: {str(e)}"
        )
