// Completion rules
hitStop = not armed and not done and ((dir == "LONG"  and low  <= lvlStop)  or (dir == "SHORT" and high >= lvlStop))
hitTp1  = not armed and not done and ((dir == "LONG"  and high >= lvlTp1)  or (dir == "SHORT" and low  <= lvlTp1))
hitTp2  = not armed and not done and ((dir == "LONG"  and high >= lvlTp2)  or (dir == "SHORT" and low  <= lvlTp2))

if hitStop
    tradeState := "Stopped"
    done := true

if hitTp2
    tradeState := "TP2 hit"
    done := true
else if hitTp1
    tradeState := "TP1 hit"

// =========================
// DRAWING: lines, labels, and position box (only on lower TF)
// =========================
var line  lnEntry = na
var line  lnStop  = na
var line  lnTp1   = na
var line  lnTp2   = na
var label lbDir   = na
var box   posBox  = na

f_reset_drawings() =>
    if not na(lnEntry)
        line.delete(lnEntry)
        lnEntry := na
    if not na(lnStop)
        line.delete(lnStop)
        lnStop := na
    if not na(lnTp1)
        line.delete(lnTp1)
        lnTp1 := na
    if not na(lnTp2)
        line.delete(lnTp2)
        lnTp2 := na
    if not na(lbDir)
        label.delete(lbDir)
        lbDir := na
    if not na(posBox)
        box.delete(posBox)
        posBox := na

// Whenever new base-day signal: clear and redraw
if barstate.isnew and showHere and not na(refDay)
    // fresh draw on first bar after state set
    f_reset_drawings()
    // Position box spans ENTRY ↔ TP2 and last N bars into the future
    if dir == "SHORT"
        posBox := box.new(left=bar_index, top=lvlEntry, right=bar_index+100, bottom=lvlTp2, bgcolor=color.new(color.red, 100 - boxOpacity), border_color=color.new(color.red, 0))
    else if dir == "LONG"
        posBox := box.new(left=bar_index, top=lvlTp2, right=bar_index+100, bottom=lvlEntry, bgcolor=color.new(color.lime, 100 - boxOpacity), border_color=color.new(color.lime, 0))
    
    colEntry = color.new(color.yellow, 0)
    colStop  = color.new(color.red, 0)
    colTp    = color.new(color.teal, 0)

    lnEntry := line.new(x1=bar_index, y1=lvlEntry, x2=bar_index+500, y2=lvlEntry, extend=extend.right, color=colEntry, width=2)
    lnStop  := line.new(x1=bar_index, y1=lvlStop,  x2=bar_index+500, y2=lvlStop,  extend=extend.right, color=colStop,  width=2)
    lnTp1   := line.new(x1=bar_index, y1=lvlTp1,  x2=bar_index+500, y2=lvlTp1,  extend=extend.right, color=color.new(color.lime, 0), width=2)
    lnTp2   := line.new(x1=bar_index, y1=lvlTp2,  x2=bar_index+500, y2=lvlTp2,  extend=extend.right, color=colTp,     width=2)

    if labelOnLevels
        label.new(bar_index, lvlEntry, "Entry: " + str.tostring(lvlEntry, format.mintick), style=label.style_label_left, color=color.new(color.yellow, 0), textcolor=color.black)
        label.new(bar_index, lvlStop,  "Stop:  " + str.tostring(lvlStop,  format.mintick), style=label.style_label_left, color=color.new(color.red, 0),    textcolor=color.white)
        label.new(bar_index, lvlTp1,   "TP1:   " + str.tostring(lvlTp1,   format.mintick), style=label.style_label_left, color=color.new(color.lime, 0),   textcolor=color.black)
        label.new(bar_index, lvlTp2,   "TP2:   " + str.tostring(lvlTp2,   format.mintick), style=label.style_label_left, color=color.new(color.teal, 0),   textcolor=color.white)

    // Direction label similar to your screenshot
    trigText = (dir == "SHORT" ? "SHORT TRIGGER" : "LONG TRIGGER")
    trigCol  = (dir == "SHORT" ? color.new(color.red, 0) : color.new(color.lime, 0))
    lbDir := label.new(bar_index, na(dir) ? close : (dir == "SHORT" ? high : low), trigText, style=label.style_label_down, color=trigCol, textcolor=color.white)

// Keep the box right edge advancing so it doesn't stop drawing
if not na(posBox)
    box.set_right(posBox, bar_index+100)
