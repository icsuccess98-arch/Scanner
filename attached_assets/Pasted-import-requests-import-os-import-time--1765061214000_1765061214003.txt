import requests
import os
import time

# ---------------------------------------------------------
# CONFIGURATION
# ---------------------------------------------------------

OANDA_KEY = os.environ["OANDA_KEY"]
OANDA_ACCOUNT = "practice"
RUN_MODE = os.environ.get("RUN_MODE", "ALL")  # DAILY, WEEKLY, MONTHLY, ALL

WEBHOOK_DAILY = os.environ["WEBHOOK_DAILY"]
WEBHOOK_WEEKLY = os.environ["WEBHOOK_WEEKLY"]
WEBHOOK_MONTHLY = os.environ["WEBHOOK_MONTHLY"]
FAIL2_FTFC_WEBHOOK = os.environ["FAIL2_FTFC_WEBHOOK"]

OANDA_URL = f"https://api-fx{OANDA_ACCOUNT}.oanda.com/v3/instruments"
HEADERS = {"Authorization": f"Bearer {OANDA_KEY}"}

# ---------------------------------------------------------
# SYMBOL LIST (Majors, Crosses, Metals, Indices, Crypto)
# ---------------------------------------------------------

OANDA_SYMBOLS = [

    # MAJORS
    "EUR_USD", "GBP_USD", "USD_JPY", "USD_CHF",
    "USD_CAD", "AUD_USD", "NZD_USD",

    # EUR CROSSES
    "EUR_GBP", "EUR_JPY", "EUR_AUD", "EUR_CAD", "EUR_CHF", "EUR_NZD",

    # GBP CROSSES
    "GBP_JPY", "GBP_AUD", "GBP_CAD", "GBP_CHF", "GBP_NZD",

    # AUD CROSSES
    "AUD_JPY", "AUD_CHF", "AUD_CAD", "AUD_NZD",

    # NZD CROSSES
    "NZD_JPY", "NZD_CHF", "NZD_CAD",

    # CHF CROSSES
    "CHF_JPY", "CHF_CAD",

    # CAD CROSS
    "CAD_JPY",

    # METALS
    "XAU_USD", "XAG_USD",

    # OIL
    "WTICO_USD",

    # INDICES
    "NAS100_USD", "US30_USD",

    # OANDA CRYPTOS
    "BTC_USD", "ETH_USD", "BCH_USD", "LTC_USD"
]

# ---------------------------------------------------------
# HELPERS
# ---------------------------------------------------------

def pretty(symbol):
    return symbol.replace("_", "/")


def candle(c):
    return {
        "open": float(c["mid"]["o"]),
        "high": float(c["mid"]["h"]),
        "low": float(c["mid"]["l"]),
        "close": float(c["mid"]["c"])
    }


def direction(c):
    return "UP" if c["close"] > c["open"] else "DOWN"


# ---------------------------------------------------------
# OANDA CANDLE FETCH
# ---------------------------------------------------------

def get_oanda_candles(symbol, granularity, count=4):
    url = f"{OANDA_URL}/{symbol}/candles"
    params = {"granularity": granularity, "count": count, "price": "M"}

    r = requests.get(url, headers=HEADERS, params=params).json()

    if "candles" not in r:
        print("OANDA ERROR:", symbol, r)
        return None

    closed = [candle(x) for x in r["candles"] if x["complete"]]

    if len(closed) < 2:
        return None

    return closed


# ---------------------------------------------------------
# SYNTHETIC Q/Y BUILDER
# ---------------------------------------------------------

def merge_candles(arr):
    return {
        "open": arr[0]["open"],
        "high": max(x["high"] for x in arr),
        "low": min(x["low"] for x in arr),
        "close": arr[-1]["close"]
    }


def build_quarter_and_year(monthlies):
    last3 = monthlies[-3:]
    last12 = monthlies[-12:]
    return merge_candles(last3), merge_candles(last12)


# ---------------------------------------------------------
# STRAT LOGIC
# ---------------------------------------------------------

def strat_type(curr, prev):
    if curr["high"] > prev["high"] and curr["low"] < prev["low"]:
        return "3"
    if curr["high"] > prev["high"] and curr["low"] >= prev["low"]:
        return "2U"
    if curr["low"] < prev["low"] and curr["high"] <= prev["high"]:
        return "2D"
    return "1"


def failed_2(curr, prev):
    s = strat_type(curr, prev)

    if s == "2U" and curr["close"] < curr["open"]:
        return "Failed 2U"

    if s == "2D" and curr["close"] > curr["open"]:
        return "Failed 2D"

    return None


# ---------------------------------------------------------
# FTFC LOGIC
# ---------------------------------------------------------

def ftfc_daily(d, w, m):
    if direction(d) == direction(w) == direction(m):
        return direction(d)
    return "MIXED"


def ftfc_weekly(w, m, q):
    if direction(w) == direction(m) == direction(q):
        return direction(w)
    return "MIXED"


def ftfc_monthly(m, q, y):
    if direction(m) == direction(q) == direction(y):
        return direction(m)
    return "MIXED"


# ---------------------------------------------------------
# MESSAGE BUILDER (NO EMPTY SECTIONS)
# ---------------------------------------------------------

def build_message(title, inside, outside, failed2u, failed2d):
    msg = f"📊 **{title} Actionables**\n"

    content_added = False

    if inside:
        msg += f"\n**{title} Inside (1)**\n"
        msg += "\n".join(f"• {i}" for i in inside)
        msg += "\n"
        content_added = True

    if outside:
        msg += f"\n**{title} Outside (3)**\n"
        msg += "\n".join(f"• {o}" for o in outside)
        msg += "\n"
        content_added = True

    if failed2u:
        msg += f"\n**{title} Failed 2U**\n"
        msg += "\n".join(f"• {f}" for f in failed2u)
        msg += "\n"
        content_added = True

    if failed2d:
        msg += f"\n**{title} Failed 2D**\n"
        msg += "\n".join(f"• {f}" for f in failed2d)
        msg += "\n"
        content_added = True

    if not content_added:
        return None

    return msg


# ---------------------------------------------------------
# SEND LOGIC
# ---------------------------------------------------------

def send_results(webhook, msg):
    if msg:
        requests.post(webhook, json={"content": msg})
        time.sleep(0.5)


# ---------------------------------------------------------
# MAIN SCAN
# ---------------------------------------------------------

def scan(title, granularity):

    inside = []
    outside = []
    failed2u = []
    failed2d = []
    ftfc_fails = []
    monthly_cache = {}

    for symbol in OANDA_SYMBOLS:

        # Monthly candles for FTFC logic
        if symbol not in monthly_cache:
            monthlies = get_oanda_candles(symbol, "M", 15)
            if not monthlies or len(monthlies) < 12:
                continue
            monthly_cache[symbol] = monthlies

        candles = get_oanda_candles(symbol, granularity, 4)
        if not candles:
            continue

        curr = candles[-1]
        prev = candles[-2]
        name = pretty(symbol)

        st = strat_type(curr, prev)
        f2 = failed_2(curr, prev)

        if st == "1":
            inside.append(name)
        elif st == "3":
            outside.append(name)

        if f2 == "Failed 2U":
            failed2u.append(name)
        elif f2 == "Failed 2D":
            failed2d.append(name)

        # Build FTFC signals
        monthlies = monthly_cache[symbol]
        q, y = build_quarter_and_year(monthlies)

        if granularity == "D":
            ftfc = ftfc_daily(curr, monthlies[-1], y)
        elif granularity == "W":
            ftfc = ftfc_weekly(curr, monthlies[-1], q)
        else:
            ftfc = ftfc_monthly(curr, q, y)

        if f2 == "Failed 2U" and ftfc == "DOWN":
            ftfc_fails.append(name)

        if f2 == "Failed 2D" and ftfc == "UP":
            ftfc_fails.append(name)

        time.sleep(0.25)

    # Build and send main actionable report
    if title == "Daily":
        msg = build_message("Daily", inside, outside, failed2u, failed2d)
        send_results(WEBHOOK_DAILY, msg)

    elif title == "Weekly":
        msg = build_message("Weekly", inside, outside, failed2u, failed2d)
        send_results(WEBHOOK_WEEKLY, msg)

    else:
        msg = build_message("Monthly", inside, outside, failed2u, failed2d)
        send_results(WEBHOOK_MONTHLY, msg)

    # FTFC failed 2 continuation report
    if ftfc_fails:
        msg = f"🚨 **FTFC Failed 2 Continuation**\n" + "\n".join(f"• {x}" for x in ftfc_fails)
        send_results(FAIL2_FTFC_WEBHOOK, msg)


# ---------------------------------------------------------
# EXECUTION BASED ON RUN_MODE
# ---------------------------------------------------------

if RUN_MODE == "DAILY" or RUN_MODE == "ALL":
    scan("Daily", "D")

if RUN_MODE == "WEEKLY" or RUN_MODE == "ALL":
    scan("Weekly", "W")

if RUN_MODE == "MONTHLY" or RUN_MODE == "ALL":
    scan("Monthly", "M")

print("DONE")
